const comparisonOperators: ComparisonOperator[] = [
  ">=",
  "<=",
  ">",
  "<",
  "==",

/**
 * 項 (Term) をパースする ("rsi(14)", "price", "70" など)
 */
function parseTerm(termStr: string): Term {
  termStr = termStr.trim();
  if (/^\d+(?:\.\d+)?$/.test(termStr)) {
    const numValue = parseFloat(termStr);
    return numValue;
  }
  return termStr;
}
 
/**
 * Term を SQL の対応する式に変換する
 */
function termToSql(term: Term, tableName: string = "ohlc"): string {
  if (typeof term === "number") {
    return term.toString();
  }
  if (typeof term === "string") {
    const funcMatch = term.match(/^([a-zA-Z_][a-zA-Z0-9_]*)\((\d+)\)$/i);
    if (funcMatch) {
      const funcName = funcMatch[1].toLowerCase();
      const arg = parseInt(funcMatch[2], 10);
      if (funcName === "rsi") {
        return `udf_rsi(close, ${arg}) OVER (ORDER BY date)`;
      } else if (funcName === "ma") {
        return `avg(close) OVER (ORDER BY date ROWS BETWEEN ${
          arg - 1
        } PRECEDING AND CURRENT ROW)`;
      } else if (funcName === "atr") {
        return `udf_atr(high, low, close, ${arg}) OVER (ORDER BY date)`;
      }
      // 他の関数もここに追加
      throw new Error(`Unknown function in term: ${term}`);
    }
 
    // ホワイトリストに基づく識別子のチェック
    const allowedIdentifiers = [
      "price",
      "high",
      "low",
      "close",
      "volume",
      "entry_price",
    ];
    const lowerTerm = term.toLowerCase(); // 識別子は大文字・小文字を区別しないという想定 (DSL仕様による)
 
    if (allowedIdentifiers.includes(lowerTerm)) {
      // "price" は "close" にマッピングする (DSL仕様の暗黙ルールかもしれないので確認が必要)
      // REQUIREMENTS.md を見ると、IDENT は price, entry_price, high, low, close, volume
      // price が close を指すという明確な記述はなさそうだが、dsl-compiler.spec.ts での挙動と合わせる
      if (lowerTerm === "price") {
        return `\`${tableName}\`.\`close\``;
      }
      return `\`${tableName}\`.\`${lowerTerm}\``; // テーブル名と列名をバッククォートで囲む
    }

    // SQLインジェクション対策：許可されていない識別子や、
    // 識別子のパターンに合致しないものはエラーとする。
    // もしくは、文字列リテラルとして扱いたい場合は明示的なクォートをDSL側で要求すべき。
    // ここでは、許可リストにない文字列は不正な識別子としてエラーを投げる。
    throw new Error(
      `Invalid identifier or unexpected string term: "${term}". ` +
        `Allowed identifiers are: ${allowedIdentifiers.join(", ")}. ` +
        `If this is a string literal, it should be quoted in the DSL or handled differently.`
    );
  }
  // Should not happen given Term type
  throw new Error(`Unknown term type: ${typeof term}`);
}
 
/**
 * 比較式文字列 ("rsi(14) > 70"など) をパースする
 */
function parseComparison(compStr: string): Compare {
  compStr = compStr.trim();
  for (const op of comparisonOperators) {
    const parts = compStr.split(op);
    if (parts.length === 2) {
      const leftStr = parts[0].trim();
      const rightStr = parts[1].trim();
      let leftTerm, rightTerm;
      try {
        leftTerm = parseTerm(leftStr);
      } catch (e: any) {
        throw new Error(
          `Failed to parse left side of comparison "${leftStr}" in "${compStr}": ${e.message}`
        );
      }
      try {
        rightTerm = parseTerm(rightStr);
      } catch (e: any) {
        throw new Error(
          `Failed to parse right side of comparison "${rightStr}" in "${compStr}": ${e.message}`
        );
      }
      return {
        left: leftTerm,
        operator: op as ComparisonOperator,
        right: rightTerm,
      };
    }
  }
  throw new Error(
    `Invalid comparison string: "${compStr}". No valid comparison operator found or incorrect structure.`
  );
}
 
/**
 * BOOL_EXPR 文字列をパースして AST (のようなもの) に変換する (簡易版)
 * "rsi(14) > 70 && ma(5) < ma(20)" のような文字列を想定
 * 括弧は演算子の優先順位には影響せず、あくまで関数の引数でのみ使用される想定
 * 現状は && のみサポート (左結合)
 */
function parseBoolExpr(expr: string): BoolExprASTNode {
  expr = expr.trim();
 
  // Initial parenthesis balance check
  let parenBalance = 0;
  for (let char of expr) {
    if (char === "(") parenBalance++;
    else if (char === ")") parenBalance--;
    if (parenBalance < 0) {
      throw new Error(
        `Mismatched parentheses: Unmatched closing parenthesis in "${expr}"`
      );
    }
  }
  if (parenBalance !== 0) {
    throw new Error(
      `Mismatched parentheses: Unmatched opening parenthesis in "${expr}"`
    );
  }
 
  // 演算子の優先順位: || < &&
  // まず、最も外側の || を見つけて分割する
  // 例: A && B || C && D  -> (A && B) || (C && D)
  // 例: (A || B) && C     -> ((A || B)) && (C) <- これは括弧があるので && が優先
 
  let balance = 0;
  let splitIndex = -1;
 
  // 1. OR (||) で分割 (括弧を考慮)
  for (let i = expr.length - 1; i >= 0; i--) {
    if (expr[i] === ")") balance++;
    else if (expr[i] === "(") balance--;
    else if (balance === 0 && expr[i] === "|" && expr[i - 1] === "|") {
      splitIndex = i - 1;
      break;
    }
  }
 
  if (splitIndex !== -1) {
    const leftExpr = expr.substring(0, splitIndex).trim();
    const rightExpr = expr.substring(splitIndex + 2).trim();
    if (!leftExpr || !rightExpr) {
      throw new Error(
        `Invalid OR expression: parts are empty near '${expr.substring(
          splitIndex,
          splitIndex + 2
        )}'`
      );
    }
    return {
      left: parseBoolExpr(leftExpr),
      operator: "||",
      right: parseBoolExpr(rightExpr),
    };
  }
 
  // 2. AND (&&) で分割 (括弧を考慮)
  balance = 0;
  splitIndex = -1;
  for (let i = expr.length - 1; i >= 0; i--) {
    if (expr[i] === ")") balance++;
    else if (expr[i] === "(") balance--;
    else if (balance === 0 && expr[i] === "&" && expr[i - 1] === "&") {
      splitIndex = i - 1;
      break;
    }
  }
 
  if (splitIndex !== -1) {
    const leftExpr = expr.substring(0, splitIndex).trim();
    const rightExpr = expr.substring(splitIndex + 2).trim();
    if (!leftExpr || !rightExpr) {
      throw new Error(
        `Invalid AND expression: parts are empty near '${expr.substring(
          splitIndex,
          splitIndex + 2
        )}'`
      );
    }
    return {
      left: parseBoolExpr(leftExpr),
      operator: "&&",
      right: parseBoolExpr(rightExpr),
    };
  }
 
  // 3. 括弧の処理
  // 式全体が括弧で囲まれている場合 (例: "(A && B)")、括弧を剥がして再評価
  if (expr.startsWith("(") && expr.endsWith(")")) {
    let openParen = 1;
    let matchingParenFound = false;
    for (let i = 1; i < expr.length - 1; i++) {
      if (expr[i] === "(") openParen++;
      else if (expr[i] === ")") openParen--;
      if (openParen === 0) {
        // これは (A) && (B) のようなケースなので、括弧を剥がすべきではない
        // (A && B) のように全体が囲まれているときのみ剥がす
        // ただし、このループは expr.length - 1 までなので、最後の ')' の直前で openParen が 0 なら
        // それは ((A)) のようなケースでも、最も外側の括弧を剥がすことになる。
        // (A) || (B) のようなケースは、上記の || で分割されるはず。
        // (A) && (B) のようなケースは、上記の && で分割されるはず。
        // よって、ここに到達するのは、単一の比較式か、(単一の比較式 or 単一の論理式) の形のはず。
        // ( (A && B) || C ) のような形であれば、さらに内部をパースする
        matchingParenFound = i === expr.length - 2; // 最後の文字の直前で閉じていれば全体を囲む括弧
        break;
      }
    }
    if (openParen === 1 && matchingParenFound) {
      // openParen が初期値の1に戻り、かつそれが文字列の最後で起きた場合
      const innerExpr = expr.substring(1, expr.length - 1).trim();
      if (!innerExpr) {
        throw new Error("Empty expression inside parentheses");
      }
      return parseBoolExpr(innerExpr);
    }
    // そうでなければ、これは ((A) && B) のようなものではなく、(A) && (B) のように演算子が括弧の外にあるケースか、
    // 単純な比較式。しかし、そのケースは上記のAND/OR分割で処理されているはず。
    // ここに来るのは、 ( A > B ) のような、既に不要な括弧で囲まれた比較式か、
    // もしくはパースエラーの可能性がある。ひとまず比較式として処理を試みる。
  }
 
  // 4. 比較式としてパース
  // ここに到達するのは、論理演算子がなく、括弧で適切に囲まれていないか、
  // または括弧が剥がされた結果の単純な比較式
  try {
    return parseComparison(expr);
  } catch (e) {
    console.error(`Failed to parse as comparison: '${expr}'`, e);
    throw new Error(
      `Invalid boolean expression structure: '${expr}'. Could not parse as logical or comparison expression.`
    );
  }
}
 
/**
 * パースされた BOOL_EXPR AST を SQL の WHERE 句の述語に変換する
 */
function astToSqlPredicate(
  astNode: BoolExprASTNode,
  tableName: string = "ohlc"
): string {
  // カスタム型ガード関数 (より明確にするため)
  function isLogicalNode(node: BoolExprASTNode): node is LogicalOperationNode {
    return (
      typeof node === "object" &&
      node !== null &&
      "operator" in node &&
      logicalOperators.includes(node.operator as LogicalOperator) &&
      "left" in node && // left と right が存在することも確認
      "right" in node
    );
  }
 
  function isCompareNode(node: BoolExprASTNode): node is Compare {
    return (
      typeof node === "object" &&
      node !== null &&
      "operator" in node &&
      comparisonOperators.includes(node.operator as ComparisonOperator) &&
      "left" in node &&
      "right" in node &&
      // Compareノードのleft/rightはTerm型なので、それがBoolExprASTNodeではないことで区別も可能だが、
      // operatorの種類で判別するのが主目的
      !logicalOperators.includes(node.operator as LogicalOperator) // 論理演算子ではないことを確認
    );
  }
 
  if (isLogicalNode(astNode)) {
    // astNode はここで LogicalOperationNode と推論される
    const leftPredicate = astToSqlPredicate(astNode.left, tableName); // as any 不要
    const rightPredicate = astToSqlPredicate(astNode.right, tableName); // as any 不要
    return `(${leftPredicate} ${astNode.operator} ${rightPredicate})`;
  } else if (isCompareNode(astNode)) {
    // astNode はここで Compare と推論される
    const leftSql = termToSql(astNode.left, tableName);
    const rightSql = termToSql(astNode.right, tableName);
    return `${leftSql} ${astNode.operator} ${rightSql}`;
  }
  // 上記のいずれでもない場合、または型ガードが不完全な場合のエラー
  // 例えば、BoolExprASTNode が将来的に別の型を持つ可能性を考慮
  const exhaustiveCheck: never = astNode; // これにより、もし astNode が予期せぬ型だった場合にコンパイルエラー
  throw new Error(`Invalid AST node structure: ${JSON.stringify(astNode)}`);
}
 
/**
 * StrategyDSL オブジェクト全体を SQL クエリに変換する (SELECT 文の骨子)
 */
export function compileDslToSql(
  dsl: StrategyDSL,
  ohlcTableName: string = "ohlc"
): string {
  const entryAst = parseBoolExpr(dsl.entry.condition);
  const entryPredicate = astToSqlPredicate(entryAst, ohlcTableName);
  // console.log("compileDslToSql called with:", dsl.entry.condition); // Log input
  // console.log("Generated Predicate:", entryPredicate);
  return `SELECT * FROM \`${ohlcTableName}\` WHERE ${entryPredicate};`;
}
